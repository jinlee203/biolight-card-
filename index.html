// BioLight 명함 DB - Google Apps Script // 탭하면 자동 복사됩니다 const SHEET_NAME = '명함 DB'; const HEADERS = ['No.','전시회명','국가','수집일자','회사명','성명','직함', '이메일','전화','휴대폰','담당사원','★등급','잠재력점수','팔로업', 'D+7 알림일','D+30 알림일','메모','저장시각']; function doPost(e) { try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); // 헤더 서식 const hRange = sheet.getRange(1, 1, 1, HEADERS.length); hRange.setBackground('#2D2D2D').setFontColor('#FFFFFF') .setFontWeight('bold').setFontSize(9); sheet.setFrozenRows(1); } const lastRow = sheet.getLastRow(); const no = lastRow; // 헤더 제외 const now = new Date().toLocaleString('ko-KR'); sheet.appendRow([ no, data.expo||'', data.country||'', data.date||'', data.company||'', data.name||'', data.title||'', data.email||'', data.phone||'', data.mobile||'', data.salesperson||'', data.grade||'B', data.score||65, data.followup||'미완료', data.d7||'', data.d30||'', data.memo||'', now ]); // 등급에 따라 배경색 const newRow = sheet.getLastRow(); if (data.grade === 'A') { sheet.getRange(newRow, 12).setBackground('#FDE8E6').setFontColor('#F03C2E'); } else if (data.grade === 'B') { sheet.getRange(newRow, 12).setBackground('#FFF4DC').setFontColor('#B86800'); } return ContentService .createTextOutput(JSON.stringify({status:'ok', row:newRow})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:'error', message:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'ok', message:'BioLight 명함 DB 연결 정상'})) .setMimeType(ContentService.MimeType.JSON); }
